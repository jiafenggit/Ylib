Function-like Macro


IS_SPACE

#define	IS_SPACE(c)	((c) == SPACE || (c) == TAB || (c) == LF || (c) == CR)
True if the character is a space.

IS_CHAR

#define IS_CHAR(c)	(((c) >= 'a' && (c) <= 'z') || ((c) >= 'A' && (c) <= 'Z'))
True if the character is an ASCII character.

IS_NUM

#define	IS_NUM(c)	((c) >= '0' && (c) <= '9')
True if the character is a numeric digit.

IS_HEXA

#define	IS_HEXA(c)	(IS_NUM(c) || ((c) >= 'a' && (c) <= 'f') || ((c) >= 'A' && (c) <= 'F'))
True if the character is an hexadecimal digit.

ARRAY_SIZE

#define ARRAY_SIZE(a)	(sizeof(a) / sizeof(a[0]))
Return the number of elements in an array.
